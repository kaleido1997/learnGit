TLDR:
	git init
	git add *
	git commit -m "blabla"
	git push (may need to pull before push)

	含分支：
	cd whereYouWantToCreateThisProject
	git init
	git remote add origin theUrl	// 将本地关联至远程repository，将远程repo简称为origin
	git branch / branch -a		// 查看本地/远程分支
	git checkout -b branchNameLocal	// 在本地创建新分支
	git status			// 查看当前文件状态（是否add、commit等）
	git add *
	git commit -m "info"
	git branch --set-upstream-to=origin/branchNameRemote branchNameLocal	// 关联本地及远程分支
	git push -u origin branchNameRemote			
	// 由于远程库是空的，我们第一次推送branchNameLocal分支时，加上了-u参数，
	// Git不但会把本地的branchNameLocal分支内容推送的远程新的branchNameRemote分支，
	// 还会把本地的branchNameLocal分支和远程的branchNameRemote分支关联起来，
	// 在以后的推送或者拉取时就可以简化命令。
	git push --set-upstream origin explore
	// 若之前更新代码时未加-u，则可之后再通过--set-upstream来关联本地及远程repo

Pull:
	cd thePlaceYouWant
	git clone theLinkOfGit
	(edit)
	git stash save "blabla"
	git pull
